# show Rent
SELECT 
	Rent.ID, 
	Car.ID as Car_ID, 
	Car.Color,
	Car.Model,
	Client.ID as Client_ID,
	Client.Name,
	Rent.Start,
	Rent.End,
	Rent.Price
FROM Rent
JOIN Car
  ON Rent.Car = Car.ID 
JOIN Client
  ON Rent.Client = Client.ID 

# show Maintenance
SELECT 
	Maintenance.ID,
	Car.ID as Car_ID,
	Service.ID as Service_ID,
	Maintenance.Start,
	Maintenance.End,
	Maintenance.Price
FROM Maintenance
JOIN Car
  ON Maintenance.Car = Car.ID 
JOIN Service
  ON Maintenance.Service = Service.ID
